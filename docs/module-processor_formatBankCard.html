processor/formatBankCard

格式化银行卡号
Source:
  • processor/formatBankCard.js, line 2
Since:
  • 1.1.0
Example
// 19位银行卡
formatBankCard('6228480402564890018');
// => 6228 4804 0256 4890 018

// 16位银行卡
formatBankCard('6228480402564890');
// => 6228 4804 0256 4890

// 16位银行卡，"-"间隔
formatBankCard('6228480402564890', {char: '-'});
// => 6228-4804-0256-4890
